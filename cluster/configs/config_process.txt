# zookeeper 

# download
wget -P /opt/package http://archive.apache.org/dist/zookeeper/zookeeper-3.5.7/apache-zookeeper-3.5.7-bin.tar.gz

mv /opt/configs/zookeeper/zoo.cfg /opt/module/zookeeper/conf/zoo.cfg
